-- Criação do banco de dados
CREATE DATABASE IF NOT EXISTS fabrica_caixas;
USE fabrica_caixas;

-- Tabela para os dados da empresa (anos e quantidade de caixas)
CREATE TABLE empresa_stats (
    id INT AUTO_INCREMENT PRIMARY KEY,
    anos_historia INT NOT NULL,
    caixas_vendidas BIGINT NOT NULL
);

-- Tabela para notícias e promoções
CREATE TABLE mural_noticias (
    id INT AUTO_INCREMENT PRIMARY KEY,
    tipo ENUM('noticia', 'promocao') NOT NULL,
    titulo VARCHAR(100) NOT NULL,
    descricao TEXT NOT NULL,
    data_publicacao DATE NOT NULL
);

-- Tabela para os Slides (Destaques, Alimentício e Industrial)
CREATE TABLE IF NOT EXISTS produtos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    secao ENUM('destaque', 'alimenticio', 'industrial') NOT NULL,
    titulo VARCHAR(100) NOT NULL,
    descricao TEXT NOT NULL,
    imagem_url VARCHAR(255) NOT NULL
);

-- Tabela para os dados da sessão "A Fábrica" (Apenas 1 registro que será sempre atualizado)
CREATE TABLE IF NOT EXISTS empresa_info (
    id INT AUTO_INCREMENT PRIMARY KEY,
    titulo VARCHAR(150) NOT NULL,
    texto_historia TEXT NOT NULL,
    img_mosaico_1 VARCHAR(255) NOT NULL,
    img_mosaico_2 VARCHAR(255) NOT NULL,
    img_mosaico_3 VARCHAR(255) NOT NULL
);

-- Tabela para os Modelos e Tamanhos do Formulário
CREATE TABLE IF NOT EXISTS form_modelos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    secao ENUM('alimenticio', 'industrial') NOT NULL,
    nome VARCHAR(100) NOT NULL,
    tamanho VARCHAR(100) NOT NULL
);

-- Tabela para os Materiais do Formulário
CREATE TABLE IF NOT EXISTS form_materiais (
    id INT AUTO_INCREMENT PRIMARY KEY,
    secao ENUM('alimenticio', 'industrial', 'ambos') NOT NULL,
    nome VARCHAR(100) NOT NULL
);
